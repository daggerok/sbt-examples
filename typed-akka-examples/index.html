# typed akka

with java use:

//...
import akka.actor.typed.*;
import akka.actor.typed.javadsl.*;
//...

with scala use:

//...
import akka.actor.typed._
import akka.actor.typed.scaladsl._
//...

./sbtw "project javaHello" clean run
./sbtw "project scalaHello" clean run
